namespace = orbital_bombardment

planet_event = {
	id = orbital_bombardment.1
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		# Kill a pop
		random_list = {
			1 = {
				if = {
					limit = {
						or = {
							orbital_bombardment = limited
							orbital_bombardment = full
							AND = {
								orbital_bombardment = soft
								fortification_health > 1  # don't kill any Pop if set to Light and fortification is at 1% or lower
							}
						}
					}
					if = {
						limit = {
							count_pops = {
								count > 1
								limit = {
									is_sentient = yes
									owner = { NOT = { is_country_type = "primitive" } }
								}
							}
						}
						FROM = {
							country_event = { id = action.23 }
						}						
						random_pop = {
							limit = {
								is_sentient = yes
								owner = { NOT = { is_country_type = "primitive" } }
							}
							kill_pop = yes
						}
						else = {
							if = {
								limit = {
									count_pops = {
										count = all
										limit = {
											OR = {
												is_sentient = no
												owner = { is_country_type = "primitive" }
											}
										}
									}
								}
								random_pop = {
									kill_pop = yes
								}								
							}
						}
					}
				}
			}
			14 = {
				if = {
					limit = {
						or = {
							orbital_bombardment = limited
							orbital_bombardment = full
						}
					}
					if = {
						limit = {
							count_pops = {
								count > 1
								limit = {
									is_sentient = yes
									owner = { NOT = { is_country_type = "primitive" } }
								}
							}
						}
						FROM = {
							country_event = { id = action.23 }
						}						
						random_pop = {
							limit = {
								is_sentient = yes
								owner = { NOT = { is_country_type = "primitive" } }
							}
							kill_pop = yes
						}
						else = {
							if = {
								limit = {
									count_pops = {
										count = all
										limit = {
											OR = {
												is_sentient = no
												owner = { is_country_type = "primitive" }
											}
										}
									}
								}
								random_pop = {
									kill_pop = yes
								}
							}
						}
					}
				}
			}
			25 = {
				if = {
					limit = {
						or = {
							orbital_bombardment = full
						}
					}
					if = {
						limit = {
							count_pops = {
								count > 1
								limit = {
									is_sentient = yes
									owner = { NOT = { is_country_type = "primitive" } }
								}
							}
						}
						FROM = {
							country_event = { id = action.23 }
						}							
						random_pop = {
							limit = {
								is_sentient = yes
								owner = { NOT = { is_country_type = "primitive" } }
							}
							kill_pop = yes
						}
						else = {
							if = {
								limit = {
									count_pops = {
										count = all
										limit = {
											OR = {
												is_sentient = no
												owner = { is_country_type = "primitive" }
											}
										}
									}
								}
								random_pop = {
									kill_pop = yes
								}
							}
						}
					}
				}
			}
			50 = {
			}
		}
		
		# Ruin a building
		random_list = {
			25 = {
				if = {
					limit = {
						or = {
							orbital_bombardment = limited
							orbital_bombardment = full
						}
					}
					random_tile = {
						limit = {
							has_building = yes
							is_ruined = no
						}
						set_ruined = yes
					}
				}
			}
			25 = {
				if = {
					limit = {
						or = {
							orbital_bombardment = full
						}
					}
					random_tile = {
						limit = {
							has_building = yes
							is_ruined = no
						}
						set_ruined = yes
					}
				}
			}
			25 = {
			}
		}

		# Create tile blocker
		random_list = {
			15 = {
				if = {
					limit = {
						orbital_bombardment = full
					}
					if = {
						limit = {
							planet = {
								OR = {
									count_pops = {
										count > 1
										limit = {
											is_sentient = yes
											owner = { NOT = { is_country_type = "primitive" } }
										}
									}
									count_pops = {
										count = all
										limit = {
											OR = {
												is_sentient = no
												owner = { is_country_type = "primitive" }
											}
										}
									}
								}
							}
						}
						random_tile = {
							limit = {
								has_blocker = no
								has_building = no
							}
							set_blocker = "tb_bombarded_land"
						}
						else = {
							random_tile = {
								limit = {
									has_blocker = no
									has_building = no
									OR = {
										NOT = { exists = pop }
										pop = {
											OR = {
												is_sentient = no
												owner = { is_country_type = "primitive" }
											}
										}
									}
								}
								set_blocker = "tb_bombarded_land"
							}
						}
					}
				}
			}
			75 = {
			}
		}
	}
}
